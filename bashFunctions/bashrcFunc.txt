function mkcd(){
	mkdir "$1"; cd "$1"
}

function up(){
	back="cd "
	if [ "$1" == "" ]; then
		back+="../"
	else
		for ((i=0; i < "$1"; i++)) do
			back+="../"	
		done
	fi
	eval "$back"
}

function almightypush(){
	if [ "$#" -eq 1 ];then
		git add -A
		git commit -m "$1"
		git push origin HEAD
	else
		echo "You need to pass a commit message in quotes \"\" as one argument"
	fi
}
